 23 	 * MUST be called to register the component
 24 	 * 
 25 	 * @param {String}
 26 	 *            name the name of the component
 27 	 * 
 28 	 * @param {config}
 29 	 *            config a config object
 30 	 * 
 31 	 * @param {String|jQuery}
 32 	 *            config.parent jquery selector, html element, jquery object (if not set, the the element will not be
 33 	 *            added in the rendering logic). <br />
 34 	 *            Note 1) If ctx.parent is absent from the component definition and from this method call, the brite
 35 	 *            will not append the returned element to the DOM. So, if ctx.parent is null, then the create() must
 36 	 *            take care of adding the elements to the DOM. However, the postDisplay will still be called.
 37 	 * @param {String}
 38 	 *            config.animation (experimental) the animation ("fromLeft" , "fromRight", or null) (default undefined)
 39 	 * @param {String|jQuery}
 40 	 *            config.replace jquery selector string, html element, or jquery object (default undefined) of the
 41 	 *            element to be replaced
 42 	 * @param {Boolean}
 43 	 *            config.emptyParent if set/true will call empty() on the parent before adding the new element (default
 44 	 *            false). Valid only if no transition and build return an element
 45 	 * @param {Boolean}
 46 	 *            config.unique if true, the component will be display only if there is not already one component with
 47 	 *            the same name in the page.
 48 	 * 
 49 	 * @param {Object|Function}
 50 	 *            componentFactory (Required) Factory function or "object template" that will be used to create the
 51 	 *            object instance. If componentFactory is a plain object, the "object template" will be cloned to create
 52 	 *            the component instance. If it is a function, it will be called and a component instance object will be
 53 	 *            exptected as return value.<br />
 54 	 *            <br />
 55 	 * 
 56 	 * A "Component" object can have the following methods <br />
 57 	 * <br />
 58 	 * component.create(data,config): (required) function that will be called with (data,config) to build the
 59 	 * component.$element.<br />
 60 	 * component.init(data,config): (optional) Will be called just after the create and the component instance has been
 61 	 * initialized. <br />
 62 	 * component.postDisplay(data,config): (optional) This method will get called with (data,config) after the component
 63 	 * has been created and initialized (postDisplay is deferred for performance optimization) <br />
 64 	 * Since this call will be deferred, it is a good place to do non-visible logic, such as event bindings.<br />
 65 	 * component.destroy() (optional) This will get called when $.bRemove or $.bEmpty is called on a parent (or on the
 66 	 * element for $.bRemove). It will get called before this component htmlElement will get removed<br />
 67 	 * component.postDestroy() (optional) This will get called when $.bRemove or $.bEmpty is called on a parent (or on
 68 	 * the element for $.bRemove). It will get called after this component htmlElement will get removed<br />
 69 	 * 
 70 	 */
 71 	brite.registerComponent = function(name, config, componentFactory) {
 72 		var def = {};
 73 		def.name = name;
 74 		def.componentFactory = componentFactory;
 75 		def.config = config;
 76 		_componentDefStore[name] = def;
 77 
 78 		var deferred = _deferredByComponentName[name];
 79 		if (deferred) {
 80 			deferred.resolve(def);
 81 			delete _deferredByComponentName[name];
 82 		}
 83 	};

160 	/**
161 	 * Return the promise().<br />
162 	 * 
163 	 * <ul>
164 	 * <li>It will load the component only if it not already loaded</li>
165 	 * 
166 	 * <li>As of now, the component is loaded by loading (via sync-AJAX class) and adding the "components/[name].html"
167 	 * content to the "body" (can be overriden with brite.config.componentsHTMLHolder)</li>
168 	 * 
169 	 * <li> So, developers need to make sure of the following:<br /> - the "component/[name].html" exists (relative to
170 	 * the current page) and does not contain any visible elements <br /> - the "component/[name].html" will call the
171 	 * briteui.registerComponent([name],componentDef) <br />
172 	 * </li>
173 	 * </ul>
174 	 * <br />
175 	 * TODO: Needs to make the the component
176 	 * 
177 	 * @param {Object}
178 	 *            name component name (no space or special character)
179 	 * @return The loaderDeferred
180 	 */
181 	function loadComponent(name) {
182 		var loaderDeferred = $.Deferred();
183 
184 		var componentDef = _componentDefStore[name];
185 
186 		// if the component already exist, just return it.
187 		if (componentDef) {
188 			loaderDeferred.resolve(componentDef);
189 		}
190 		// if the component is not loaded, load it
191 		else {
192 			var currentLoaderDeferred = _deferredByComponentName[name];
193 			// if there is already a loading going on, replace the loaderDeferred to be returned by the
194 			// currentLoaderDeferred
195 			if (currentLoaderDeferred) {
196 				loaderDeferred = currentLoaderDeferred;
197 			}
198 			// otherwise, do the loading
199 			else {
200 				_deferredByComponentName[name] = loaderDeferred;
201 				$.ajax({
202 					url : brite.config.componentsPath + name + ".html",
203 					async : true
204 				}).complete(function(jqXHR, textStatus) {
205 					$(brite.config.componentsHTMLHolder).append(jqXHR.responseText);
206 				});
207 			}
208 		}
209 		return loaderDeferred.promise();
210 	}

213 	// if $element exist, then, bypass the create
214 	function process(name, data, config, $element) {
215 		var loaderDeferred = loadComponent(name);
216 
217 		var processDeferred = $.Deferred();
218 
219 		var createDeferred = $.Deferred();
220 		var initDeferred = $.Deferred();
221 		var postDisplayDeferred = $.Deferred();
222 
223 		var processPromise = processDeferred.promise();
224 		processPromise.whenCreate = createDeferred.promise();
225 		processPromise.whenInit = initDeferred.promise();
226 		processPromise.whenPostDisplay = postDisplayDeferred.promise();
227 
228 		loaderDeferred.done(function(componentDef) {
229 			config = buildConfig(componentDef, config);
230 			var component = instantiateComponent(componentDef);
231 
232 			// If the config.unique is set, and there is a component with the same name, we resolve the deferred now
233 			// NOTE: the whenCreate and whenPostDisplay won't be resolved again
234 			// TODO: an optimization point would be to add a "bComponentUnique" in the class for data-bcomponent that
235 			// have a confi.unique = true
236 			// This way, the query below could be ".bComponentUnique [....]" and should speedup the search significantly
237 			// on UserAgents that supports the getElementsByClassName
238 			if (config.unique) {
239 				var $component = $("[data-bcomponent='" + name + "']");
240 				if ($component.length > 0) {
241 					component = $component.bComponent();
242 					processDeferred.resolve(component);
243 					return processDeferred;
244 				}
245 			}
246 
247 			// ------ create ------ //
248 			var deferred$element = $.Deferred();
249 			// if there is no element, we invoke the build
250 			if (!$element) {
251 				// Ask the component to create the new $element
252 				var createReturn = invokeCreate(component, data, config);
253 				// if it custom Deferred, then, assume it will get resolved with the $element (as by the API contract)
254 				if (createReturn && $.isFunction(createReturn.promise) && !createReturn.jquery) {
255 					// FIXME: will need to use the new jQuery 1.6 pipe here (right now, just trigger on done
256 					createReturn.done(function($element) {
257 						deferred$element.resolve($element);
258 					}).fail(function() {
259 						deferred$element.reject();
260 					});
261 				}
262 				// otherwise, if the $element is returned , resolve the deferred$element immediately
263 				else {
264 					if (createReturn) {
265 						$element = $(createReturn);
266 					}
267 					deferred$element.resolve($element);
268 				}
269 			}
270 			// if the $element is already here, then, it is an attach, so, do a immediate Deffered
271 			else {
272 				deferred$element.resolve($element);
273 			}
274 			// ------ /create ------ //
275 
276 			// ------ render & resolve ------ //
277 			deferred$element.promise().done(function($element) {
278 				// if there is an element, then, manage the rendering logic.
279 				if ($element) {
280 					// make sure we get the jQuery object
281 					$element = $($element);
282 
283 					bind$element($element, component, data, config);
284 
285 					// attached the componentPromise to this $element, this way, during rendering sub component can sync
286 					// with it.
287 					$element.data("componentProcessPromise", processPromise);
288 
289 					createDeferred.resolve(component);
290 
291 					$.when(invokeInit(component, data, config)).done(function() {
292 						// render the element
293 						// TODO: implement deferred for the render as well.
294 						renderComponent(component, data, config);
295 
296 						initDeferred.resolve(component);
297 
298 					});
299 
300 				} else {
301 					// TODO: need to look if we need this. Basically, that allow to have create methods that do/return
302 					// nothing but still instantiate the component
303 					createDeferred.resolve(component);
304 
305 					// TODO: probably need to invokeInit in thi scase as well. For now, just resolve the initDeferred
306 					initDeferred.resolve(component);
307 
308 				}
309 
310 				processPromise.whenInit.done(function() {
311 					var parentComponentProcessPromise, invokePostDisplayDfd;
312 
313 					// if there is a parent component, then need to wait until it display to display this one.
314 					if ($element && $element.parent()) {
315 						var parentComponent$Element = $element.parent().closest("[data-bcomponent]");
316 
317 						if (parentComponent$Element.length > 0) {
318 							parentComponentProcessPromise = parentComponent$Element.data("componentProcessPromise");
319 							parentComponentProcessPromise.whenPostDisplay.done(function() {
320 								invokePostDisplayDfd = invokePostDisplay(component, data, config);
321 								invokePostDisplayDfd.done(function() {
322 									postDisplayDeferred.resolve(component);
323 								});
324 							});
325 						}
326 					}
327 
328 					// if we did not have any parentComponentProcessPromise, then, just invoke
329 					if (!parentComponentProcessPromise) {
330 						invokePostDisplayDfd = invokePostDisplay(component, data, config);
331 						invokePostDisplayDfd.done(function() {
332 							postDisplayDeferred.resolve(component);
333 						});
334 					}
335 
336 				});
337 
338 			});
339 			// ------ /render & resolve ------ //
340 			// console.log
341 			processPromise.whenPostDisplay.done(function() {
342 				processDeferred.resolve(component);
343 			});
344 		});
345 
346 		return processPromise;
347 	}

349 	function renderComponent(component, data, config) {
350 
351 		if (config.transition) {
352 			var transition = brite.getTransition(config.transition);
353 
354 			if (transition) {
355 				transition(component, data, config);
356 			} else {
357 				brite.log.error("Transition [" + config.transition + "] not found. Transitions need to be registered via brite.registerTranstion(..) before call.");
358 			}
359 		}
360 		// if no transition remove/show
361 		else {
362 			if (config.replace) {
363 				$(config.replace).bRemove();
364 			}
365 
366 			// note: if there is no parent, then, the sUI.diplay caller is reponsible to add it
367 			if (config.parent) {
368 				if (config.emptyParent) {
369 					$(config.parent).bEmpty();
370 				}
371 				$(config.parent).append(component.$element);
372 			}
373 		}

386 	function instantiateComponent(componentDef) {
387 		var component;
388 		var componentFactory = componentDef.componentFactory;
389 		if (componentFactory) {
390 			// if it is a function, call it, it should return a new component object
391 			if ($.isFunction(componentFactory)) {
392 				component = componentFactory();
393 			}
394 			// if it is a plainObject, then, we clone it (NOTE: We do a one level clone)
395 			else if ($.isPlainObject(componentFactory)) {
396 				component = $.extend({}, componentFactory);
397 			} else {
398 				brite.log.error("Invalid ComponentFactory for component [" + componentDef.componentName
399 						+ "]. Only types Function or Object are supported as componentFactory. Empty component will be created.");
400 			}
401 		} else {
402 			brite.log.error("No ComponentFactory for component [" + componentDef.componentName + "]");
403 		}
404 
405 		if (component) {
406 			component.name = componentDef.name;
407 		}
408 		return component;
409 	}

436 	function invokePostDisplay(component, data, config) {
437 		var invokeDfd = $.Deferred();
438 
439 		// Call the eventual postDisplay
440 		// (differing for performance)
441 		if (component.postDisplay) {
442 			// if the component has a delay >= 0, then, we use a setTimeout
443 			if (config.postDisplayDelay >= 0) {
444 				setTimeout(function() {
445 					var postDisplayDfd = component.postDisplay(data, config);
446 					if (postDisplayDfd && $.isFunction(postDisplayDfd)) {
447 						postDisplayDfd.done(function() {
448 							invokeDfd.resolve();
449 						});
450 					} else {
451 						invokeDfd
452 					}
453 				}, config.postDisplayDelay);
454 			}
455 			// otherwise, we call it in sync
456 			else {
457 
458 				var postDisplayDfd = component.postDisplay(data, config);
459 				if (postDisplayDfd && $.isFunction(postDisplayDfd)) {
460 					postDisplayDfd.done(function() {
461 						invokeDfd.resolve();
462 					});
463 				} else {
464 					invokeDfd.resolve();
465 				}
466 			}
467 		}
468 		// if there is now postDisplay, then, trigger it anyway
469 		else {
470 			invokeDfd.resolve();
471 		}
472 
473 		return invokeDfd.promise();
474 	}

490 	/**
491 	 * 
492 	 * Return the component that this html element belong to. Thi traverse the tree backwards (this html element up to
493 	 * document) to find the closest html element containing the brite component for this name.
494 	 * 
495 	 * If a componentName is given then it will try to find the given component.
496 	 * 
497 	 * If no componentName is given, then it will return the first component found.
498 	 * 
499 	 * For example:
500 	 * 
501 	 * @example var myComponent = $(thisDiv).bComponent("myComponent");
502 	 * 
503 	 * @param {String}
504 	 *            componentName The component name to be match when traversing the tree, if undefined, then, the
505 	 *            closestComponent will be return.
506 	 * 
507 	 */
508 	$.fn.bComponent = function(componentName) {
509 
510 		// iterate and process each matched element
511 		var $componentElement;
512 		if (componentName) {
513 			$componentElement = $(this).closest("[data-bcomponent='" + componentName + "']");
514 		} else {
515 			$componentElement = $(this).closest("[data-bcomponent]");
516 		}
517 
518 		return $componentElement.data("component");
519 
520 	};

522 	/**
523 	 * Get the list of components that this htmlElement contains.
524 	 * 
525 	 * @param {string}
526 	 *            componentName (optional) if present, will filter only the component with this matching name
527 	 * @return a javascript array of all the match component
528 	 */
529 	$.fn.bFindComponents = function(componentName) {
530 		var childrenComponents = [];
531 
532 		this.each(function() {
533 			var $this = $(this);
534 
535 			var $componentElements;
536 
537 			if (componentName) {
538 				$componentElements = $(this).find("[data-bcomponent='" + componentName + "']");
539 			} else {
540 				$componentElements = $(this).find("[data-bcomponent]");
541 			}
542 
543 			$componentElements.each(function() {
544 				var $component = $(this);
545 				childrenComponents.push($component.data("component"));
546 			});
547 		});
548 
549 		return childrenComponents;
550 	}

582 	/**
583 	 * Safely empty a HTMLElement of its children HTMLElement and bComponent by calling the preRemove and postRemove on
584 	 * every child components.
585 	 * 
586 	 * @return the jQuery object
587 	 */
588 	$.fn.bEmpty = function() {
589 		return this.each(function() {
590 			var $this = $(this);
591 
592 			var componentChildren = $this.bFindComponents();
593 
594 			// call the preRemoves
595 			$.each(componentChildren, function(idx, childComponent) {
596 				processDestroy(childComponent);
597 			});
598 
599 			// do the empty
600 			$this.empty();
601 
602 			// call the postRemoves
603 			$.each(componentChildren, function(idx, childComponent) {
604 				processPostDestroy(childComponent);
605 			});
606 		});
607 	}

609 	/**
610 	 * Safely remove a HTMLElement and the related bComponent by calling the preRemote and postRemove on every child
611 	 * components as well as this component.
612 	 * 
613 	 * @return what a jquery.remove would return
614 	 */
615 	$.fn.bRemove = function() {
616 
617 		return this.each(function() {
618 			var $this = $(this);
619 			$this.bEmpty();
620 
621 			if ($this.is("[data-bcomponent]")) {
622 				var component = $this.data("component");
623 				processDestroy(component);
624 
625 				$this.remove();
626 
627 				processPostDestroy(component);
628 			} else {
629 				$this.remove();
630 			}
631 		});
632 
633 	}

640 			var destroyFunc = component.destroy || component.preRemove;
641 
642 			if ($.isFunction(destroyFunc)) {
643 				destroyFunc.call(component);
644 			}
645 		}
646 	}

653 			var postDestoryFunc = component.postDestroy || component.postRemove;
654
